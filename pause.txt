// ставити на паузу таймер
let timeoutId;
let startTime;
let remainingTime = 5000;

const element = document.querySelector('.animated-element');

function startTimeout() {
    startTime = Date.now();
    timeoutId = setTimeout(() => {
        element.remove();
    }, remainingTime);
}

element.addEventListener('mouseover', () => {
    clearTimeout(timeoutId);
    remainingTime -= Date.now() - startTime;
    element.style.animationPlayState = 'paused';
});

element.addEventListener('mouseout', () => {
    element.style.animationPlayState = 'running';
    startTimeout();
});

startTimeout();








// ставити на паузу анімацію
const element = document.querySelector('.dom-notify');

element.addEventListener('mouseover', () => {
	element.querySelector('.dom-notify-timeline-bar-progress').style.animationPlayState = 'paused';
	
});

element.addEventListener('mouseout', () => {
	element.querySelector('.dom-notify-timeline-bar-progress').style.animationPlayState = 'running';
});
